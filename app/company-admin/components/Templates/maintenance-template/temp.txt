function handleFilterChange({
    field,
    value,
    from,
    to,
}: {
    field: string;
    value?: string;
    from?: string;
    to?: string;
}) {
    setDraftFilters((prev) => {
        const existingIndex = prev.findIndex((f) => f.field === field);

        const existing = existingIndex !== -1 ? prev[existingIndex] : {};

        const updatedFilter: DraftFilter = {
            ...existing,
            field,
            ...(value !== undefined && { text: value }),
            ...(from !== undefined && { from }),
            ...(to !== undefined && { to }),
        };

        // Delete: remove if all values cleared
        const isEmpty = !updatedFilter.text && !updatedFilter.from && !updatedFilter.to;
        if (isEmpty) {
            return prev.filter((f) => f.field !== field);
        }

        // Update or Create
        if (existingIndex !== -1) {
            const updated = [...prev];
            updated[existingIndex] = updatedFilter;
            return updated;
        }

        return [...prev, updatedFilter];
    });
}

function resolveCondition(draft: DraftFilter): Filter | null {
    const hasFrom = !!draft.from;
    const hasTo = !!draft.to;

    if (hasFrom && hasTo) {
        return {
            field: draft.field,
            condition: "between",
            text: [draft.from!, draft.to!],
        };
    }

    if (hasFrom) {
        return {
            field: draft.field,
            condition: "gte",
            text: draft.from!,
        };
    }

    if (hasTo) {
        return {
            field: draft.field,
            condition: "lte",
            text: draft.to!,
        };
    }

    if (draft.text) {
        return {
            field: draft.field,
            condition: "contains",
            text: draft.text,
        };
    }

    return null;
}

const FilterField = React.memo(function ({ colName, type, value, onFilterChange }: FilterFieldProps) {
    // Always expect value to be string[] for DATETIME, string for others
    const handleChange = (e: React.ChangeEvent<HTMLInputElement>, index?: number) => {
        if (type === "DATETIME" && index !== undefined) {
            const updated = [...(value as string[])]; // always array
            updated[index] = e.target.value;
            onFilterChange(colName, type, updated);
        } else {
            onFilterChange(colName, type, e.target.value);
        }
    };

    // Normalize dateValues for UI
    const dateValues: string[] =
        type === "DATETIME" ? ((value as string[]).length === 2 ? (value as string[]) : ["", ""]) : [];

    return (
        <div className={type === "DATETIME" ? "col-span-3" : "col-span-1"}>
            <label>{colName.replace(/_/g, " ").replace(/\b\w/g, (c) => c.toUpperCase())}</label>
            {type === "DATETIME" ? (
                <div className='grid grid-cols-2 gap-2'>
                    <input
                        type='date'
                        className='form-input'
                        onChange={(e) => handleChange(e, 0)}
                        value={dateValues[0] || ""}
                    />
                    <input
                        type='date'
                        className='form-input'
                        onChange={(e) => handleChange(e, 1)}
                        value={dateValues[1] || ""}
                    />
                </div>
            ) : type === "INTEGER" || type === "BIGINT" ? (
                <input type='number' className='form-input' value={value as string} onChange={handleChange} />
            ) : (
                <input type='text' className='form-input' value={value as string} onChange={handleChange} />
            )}
        </div>
    );
});
